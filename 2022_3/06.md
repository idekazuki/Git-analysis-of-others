```
import math

N, K = map(int, input().split())

A = list(map(int, input().split()))

ok = N-1
ng = 0

if A[N-1] < K:
    print(-1)
else:
    while(abs(ng - ok) > 1):
        half = (ng + ok) // 2
        print(half)
        if A[half] >= K:
            ok = half
        else:
            ng = half

    print(f"ok {ok} ng {ng}")
```
