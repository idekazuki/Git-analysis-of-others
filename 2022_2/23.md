data = list(map(int, input().split()))
direction = input()

news = "NEWS"

swap_position = [(1, 5, 2, 3, 0, 4),#N
                 (3, 1, 2, 5, 0, 4),#E
                 (2, 1, 5, 0, 4, 3),#W
                 (4, 0, 2, 3, 5, 1),#S
                 ]

for d in direction:
    data[:] = (data[e] for e in swap_position[news.index(d)])

print(data[0])
